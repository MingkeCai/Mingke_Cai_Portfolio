# Define the Stacking Ensemble Class
class StackingEnsemble(BaseEstimator, RegressorMixin):
    def __init__(self, base_models, meta_model, n_folds=5):
        self.base_models = base_models
        self.meta_model = meta_model
        self.n_folds = n_folds
        self.feature_names_ = None
        
    def fit(self, X, y):
        self.feature_names_ = X.columns.tolist()
        self.base_models_ = [list() for _ in self.base_models]
        oof_predictions = np.zeros((X.shape[0], len(self.base_models)))
        kf = KFold(n_splits=self.n_folds, shuffle=True, random_state=42)
        
        for i, model in enumerate(self.base_models):
            for train_index, val_index in kf.split(X):
                instance = clone(model)
                
                # Special handling for XGBoost models
                if isinstance(instance, XGBRegressor):
                    # Create validation set from training data
                    X_train_fold = X.iloc[train_index]
                    y_train_fold = y.iloc[train_index]
                    # Use 20% of training fold as validation
                    X_train_sub, X_val_sub, y_train_sub, y_val_sub = train_test_split(
                        X_train_fold, y_train_fold, 
                        test_size=0.2, 
                        random_state=42
                    )
                    instance.fit(
                        X_train_sub, 
                        y_train_sub,
                        eval_set=[(X_val_sub, y_val_sub)],
                        verbose=False
                    )
                else:
                    # Normal fitting for other models
                    instance.fit(X.iloc[train_index], y.iloc[train_index])
                    
                self.base_models_[i].append(instance)
                oof_predictions[val_index, i] = instance.predict(X.iloc[val_index])
        
        # Train the meta-model on the out-of-fold predictions
        self.meta_model_ = clone(self.meta_model)
        self.meta_model_.fit(oof_predictions, y)
        return self

    def predict(self, X):
        meta_features = np.column_stack([
            np.column_stack([model.predict(X) for model in base_models]).mean(axis=1)
            for base_models in self.base_models_
        ])
        return self.meta_model_.predict(meta_features)

    def get_feature_importance(self):
        """
        Get feature importance from the meta-model and combine with base model predictions,
        normalized between 0 and 1
        """
        # Get feature importance from meta-model
        meta_importance = self.meta_model_.feature_importances_
        
        # Create a dictionary to store combined feature importance
        combined_importance = {}
        
        # Iterate through base models and their corresponding meta-model weights
        for i, base_models in enumerate(self.base_models_):
            # Average feature importance across folds for each base model
            model_importance = np.mean([model.feature_importances_ for model in base_models], axis=0)
            
            # Get feature names
            if hasattr(base_models[0], 'feature_names_'):
                feature_names = base_models[0].feature_names_
            else:
                feature_names = X_combined.columns  # Using your existing X_combined
                
            # Scale importance by meta-model importance for this base model
            scaled_importance = model_importance * meta_importance[i]
            
            # Add to combined importance
            for feat_name, importance in zip(feature_names, scaled_importance):
                if feat_name in combined_importance:
                    combined_importance[feat_name] += importance
                else:
                    combined_importance[feat_name] = importance
        
        # Convert to DataFrame
        final_importance = pd.DataFrame({
            'Feature': list(combined_importance.keys()),
            'Importance': list(combined_importance.values())
        })
        
        # Sort by importance and get top 20
        final_importance = final_importance.sort_values('Importance', ascending=False).head(20)
        
        # # Normalize importance scores between 0 and 1
        # final_importance['Importance'] = (final_importance['Importance'] - final_importance['Importance'].min()) / \
        #                                (final_importance['Importance'].max() - final_importance['Importance'].min())
        
        return final_importance

def analyze_ensemble_importance(ensemble_model) -> Tuple[pd.DataFrame, list]:
    """
    Analyze and plot feature importance from ensemble model with scaled values
    """
    # Get feature importance
    importance_df = ensemble_model.get_feature_importance()
    
    # Scale importance values to be between 0 and 0.15 (similar to Random Forest scale)
    max_importance = importance_df['Importance'].max()
    importance_df['Importance'] = importance_df['Importance'] * (0.15 / max_importance)
    
    # Store top 20 feature names
    top_features = importance_df['Feature'].tolist()
    
    # Create the plot with white background
    plt.figure(figsize=(12, 8))
    plt.style.use('default')
    
    # Create horizontal bar plot
    bars = plt.barh(importance_df['Feature'], 
                   importance_df['Importance'],
                   color='#3182bd',  # Steel blue color
                   height=0.7)
    
    # # Add value labels on the bars
    # for i, bar in enumerate(bars):
    #     width = bar.get_width()
    #     plt.text(width * 1.02,  # Slight offset from end of bar 
    #             bar.get_y() + bar.get_height()/2,
    #             f'{width:.4f}',
    #             ha='left', 
    #             va='center',
    #             fontsize=9)
    
    # Customize the plot
    plt.xlabel('Importance')
    plt.ylabel('Features')
    plt.title('Top 20 Feature Importance (Ensemble Model)', pad=20, size=12)
    
    # Remove top and right spines
    plt.gca().spines['top'].set_visible(False)
    plt.gca().spines['right'].set_visible(False)
    
    # Add thin gridlines
    plt.grid(axis='x', linestyle='--', alpha=0.3)

    plt.xlim(0, importance_df['Importance'].max() * 1.15)
    
    # Format axis labels

    # Adjust layout to prevent label cutoff
    plt.tight_layout()
    
    # Invert Y-axis to show most important features at top
    plt.gca().invert_yaxis()
    
    plt.show()
    
    return importance_df, top_features

def create_simplified_model(X_combined, y_rf, top_20_features):
    """
    Create and evaluate a simplified linear regression model using top features
    
    Parameters:
    -----------
    X_combined : pd.DataFrame
        Complete feature dataset
    y_rf : pd.Series
        Target variable (paper quality)
    top_20_features : list
        List of top 20 feature names from ensemble model
    
    Returns:
    --------
    tuple: (model coefficients, feature names, model metrics)
    """
    # Select only the top 20 features
    X_simplified = X_combined[top_20_features]
    
    # Split the data
    X_train, X_test, y_train, y_test = train_test_split(
        X_simplified, y_rf, test_size=0.2, random_state=42
    )
    
    # Scale the features
    scaler = StandardScaler()
    X_train_scaled = scaler.fit_transform(X_train)
    X_test_scaled = scaler.transform(X_test)
    
    # Train linear regression model
    lr_model = LinearRegression()
    lr_model.fit(X_train_scaled, y_train)
    
    # Make predictions
    y_pred = lr_model.predict(X_test_scaled)
    
    # Calculate metrics
    r2 = r2_score(y_test, y_pred)
    mse = mean_squared_error(y_test, y_pred)
    rmse = np.sqrt(mse)
    mae = mean_absolute_error(y_test, y_pred)
    
    # Create coefficient dataframe
    coefficients_df = pd.DataFrame({
        'Feature': top_20_features,
        'Coefficient': lr_model.coef_,
        'Scaled_Mean': scaler.mean_,
        'Scaled_Scale': scaler.scale_
    })
    
    # Sort by absolute coefficient value
    coefficients_df['Abs_Coefficient'] = abs(coefficients_df['Coefficient'])
    coefficients_df = coefficients_df.sort_values('Abs_Coefficient', ascending=False)
    
    # Print model performance
    print("\nLinear Regression Model Performance Metrics:")
    print("-" * 40)
    print(f"Mean Squared Error (MSE)       : {mse:.6f}")
    print(f"Root Mean Squared Error (RMSE) : {rmse:.6f}")
    print(f"Mean Absolute Error (MAE)      : {mae:.6f}")
    print(f"R-squared (R²)                 : {r2:.6f}")
    
    # Plot actual vs predicted values
    plt.figure(figsize=(10, 6))
    plt.scatter(y_test, y_pred, alpha=0.5)
    plt.plot([y_test.min(), y_test.max()], [y_test.min(), y_test.max()], 'r--', lw=2)
    plt.xlabel('Actual Paper Quality')
    plt.ylabel('Predicted Paper Quality')
    plt.title('Actual vs Predicted Paper Quality\nSimplified Linear Model')
    plt.grid(True, alpha=0.3)
    
    # Add performance metrics to plot
    metrics_text = f'R² = {r2:.3f}\nMSE = {mse:.3f}\nRMSE = {rmse:.3f}\nMAE = {mae:.3f}'
    plt.text(0.05, 0.95, metrics_text,
             transform=plt.gca().transAxes,
             verticalalignment='top',
             bbox=dict(boxstyle='round', facecolor='white', alpha=0.8))
    
    plt.tight_layout()
    plt.show()
    
    # Generate the formula
    print("\nPaper Quality Prediction Formula:")
    print("PaperQuality = ", end="")
    
    formula_terms = []
    for _, row in coefficients_df.iterrows():
        # Calculate the actual coefficient considering scaling
        actual_coef = row['Coefficient'] / row['Scaled_Scale']
        offset = -row['Coefficient'] * row['Scaled_Mean'] / row['Scaled_Scale']
        
        formula_terms.append(f"({actual_coef:.6f} × {row['Feature']})")
        if offset != 0:
            formula_terms.append(f"({offset:.6f})")
    
    print(" + ".join(formula_terms))
    print(f"\nIntercept: {lr_model.intercept_:.6f}")
    
    return coefficients_df, lr_model.intercept_, {
        'mse': mse,
        'rmse': rmse,
        'mae': mae,
        'r2': r2,
    }

def generate_simplified_formula(coefficients_df, intercept):
    """
    Generate a simplified version of the prediction formula
    """
    print("\nSimplified Paper Quality Prediction Formula:")
    print("PaperQuality = ", end="")
    
    # Combine coefficients and offsets
    combined_terms = []
    total_constant = intercept
    
    for _, row in coefficients_df.iterrows():
        # Calculate actual coefficient
        actual_coef = row['Coefficient'] / row['Scaled_Scale']
        # Add offset to total constant
        total_constant += -row['Coefficient'] * row['Scaled_Mean'] / row['Scaled_Scale']
        # Add the term
        combined_terms.append(f"({actual_coef:.6f} × {row['Feature']})")
    
    # Print formula with a single constant term at the end
    print(" + ".join(combined_terms))
    print(f"\nConstant Term: {total_constant:.6f}")
    
    return combined_terms, total_constant

def analyze_feature_directions(data: pd.DataFrame, top_features: list) -> pd.DataFrame:
    """
    Analyze how each top feature relates to paper quality and provide recommendations
    
    Parameters:
    -----------
    data : pd.DataFrame
        Original dataset with features and quality measure
    top_features : list
        List of top 20 feature names
    
    Returns:
    --------
    pd.DataFrame:
        Analysis results including correlation and recommended direction
    """
    # Calculate correlations with quality measure
    selected_data = data[top_features + ['QualityMeasure']]
    correlations = selected_data.corr()['QualityMeasure'][:-1]  # Exclude self-correlation
    
    # Create analysis DataFrame
    analysis_df = pd.DataFrame({
        'Feature': top_features,
        'Correlation': correlations,
        'Impact_Strength': abs(correlations),
        'Recommended_Direction': ['Increase' if corr > 0 else 'Decrease' for corr in correlations]
    })
    
    # Sort by absolute correlation strength
    analysis_df = analysis_df.sort_values('Impact_Strength', ascending=False)
    
    # Add interpretation column
    analysis_df['Interpretation'] = analysis_df.apply(
        lambda row: f"{'Increase' if row['Correlation'] > 0 else 'Decrease'} "
                   f"{row['Feature']} to improve paper quality "
                   f"(correlation strength: {abs(row['Correlation']):.3f})",
        axis=1
    )
    
    # Create visualization
    plt.figure(figsize=(12, 8))
    colors = ['#2ecc71' if c > 0 else '#e74c3c' for c in analysis_df['Correlation']]
    
    bars = plt.barh(analysis_df['Feature'], 
                   analysis_df['Correlation'],
                   color=colors,
                   height=0.7)
    
    # Add value labels
    for i, bar in enumerate(bars):
        width = bar.get_width()
        plt.text(width * (1.02 if width > 0 else 0.98),
                bar.get_y() + bar.get_height()/2,
                f'{width:.3f}',
                ha='left' if width > 0 else 'right',
                va='center',
                fontsize=9)
    
    plt.axvline(x=0, color='black', linestyle='-', linewidth=0.5)
    plt.xlabel('Correlation with Paper Quality')
    plt.ylabel('Features')
    plt.title('Feature Correlations with Paper Quality\nGreen = Positive Impact, Red = Negative Impact', 
             pad=20, size=12)
    
    # Remove top and right spines
    plt.gca().spines['top'].set_visible(False)
    plt.gca().spines['right'].set_visible(False)
    
    # Add thin gridlines
    plt.grid(axis='x', linestyle='--', alpha=0.3)
    
    # Adjust layout
    plt.tight_layout()
    
    # Invert Y-axis to match importance plot
    plt.gca().invert_yaxis()
    
    plt.show()
    
    return analysis_df
